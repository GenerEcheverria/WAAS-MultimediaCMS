      email: ['', [Validators.required, Validators.email]],
      tel: ['', [Validators.required, Validators.pattern('^((\\+52-?)|0)?[0-9]{10}$')]],

      cpass: ['', [Validators.required, Validators.maxLength(32), Validators.minLength(8)]]

  }
